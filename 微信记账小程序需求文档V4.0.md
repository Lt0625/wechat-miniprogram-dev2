### 2.2 账户自动余额 ⭐⭐⭐⭐⭐

#### 2.2.1 功能概述

根据用户的账单记录，自动计算每个账户的实时余额，无需手动维护。

#### 2.2.2 计算逻辑

```
账户余额计算公式：

对于每个账户 A：
  A.余额 = Σ(收入账单金额) - Σ(支出账单金额)
  
  其中：
  - 收入账单：type='income' AND account='A'
  - 支出账单：type='expense' AND account='A'
```

#### 2.2.5 技术实现

```javascript
// 账户余额计算模块
function calculateAccountBalance(bills, accountName) {
  let income = 0
  let expense = 0

  bills.forEach(bill => {
    if (bill.account !== accountName) return
    
    if (bill.type === 'income') {
      income += parseFloat(bill.amount) || 0
    } else {
      expense += parseFloat(bill.amount) || 0
    }
  })

  return {
    balance: income - expense,
    income,
    expense
  }
}

// 资产页数据加载
loadAccounts() {
  const bills = app.getBillsFromCache()
  const accounts = wx.getStorageSync('accounts') || []
  
  const accountsWithBalance = accounts.map(acc => {
    const { balance, income, expense } = calculateAccountBalance(bills, acc.name)
    return {
      ...acc,
      balance,
      income,
      expense
    }
  })
  
  // 计算汇总
  const totalAssets = accountsWithBalance
    .filter(acc => acc.balance >= 0)
    .reduce((sum, acc) => sum + acc.balance, 0)
    
  const totalLiability = accountsWithBalance
    .filter(acc => acc.balance < 0)
    .reduce((sum, acc) => sum + Math.abs(acc.balance), 0)
    
  const netAssets = totalAssets - totalLiability
  
  this.setData({
    accounts: accountsWithBalance,
    totalAssets,
    totalLiability,
    netAssets
  })
}
```

```javascript
// 数据迁移：给已有账户补充余额信息
migrateAccountBalances() {
  const bills = app.getBillsFromCache()
  const accounts = wx.getStorageSync('accounts') || []
  
  // 检查账户是否有balance字段
  const accountsNeedMigration = accounts.filter(acc => acc.balance === undefined)
  
  if (accountsNeedMigration.length > 0) {
    wx.showModal({
      title: '数据迁移',
      content: `检测到${accountsNeedMigration.length}个账户需要计算余额，是否立即计算？`,
      success: (res) => {
        if (res.confirm) {
          const migratedAccounts = accounts.map(acc => {
            if (acc.balance === undefined) {
              const { balance } = calculateAccountBalance(bills, acc.name)
              return { ...acc, balance }
            }
            return acc
          })
          wx.setStorageSync('accounts', migratedAccounts)
          wx.showToast({ title: '迁移完成', icon: 'success' })
        }
      }
    })
  }
}
```

### 2.4 重复检测 ⭐⭐⭐⭐⭐

#### 2.4.1 功能概述

保存账单前检测是否有重复记录，防止误记。

#### 2.4.2 检测规则

```javascript
function detectDuplicate(newBill, existingBills) {
  const today = newBill.date
  const sameDayBills = existingBills.filter(b => b.date === today)
  
  // 检测条件：同日期 + 同分类 + 同金额 + 同账户
  const duplicate = sameDayBills.find(b => 
    b.category === newBill.category &&
    b.amount === newBill.amount &&
    b.account === newBill.account
  )
  
  return duplicate
}
```

```javascript
// 保存前检测重复
saveBill() {
  const newBill = this.prepareBillData()
  
  // 检测重复
  const bills = wx.getStorageSync('bills') || []
  const duplicate = detectDuplicate(newBill, bills)
  
  if (duplicate) {
    wx.showModal({
      title: '⚠️ 检测到相似账单',
      content: `今天已有一笔 ${duplicate.category} ¥${duplicate.amount}，是否继续保存？`,
      confirmText: '继续保存',
      cancelText: '取消',
      success: (res) => {
        if (res.confirm) {
          this.doSaveBill(newBill)
        }
      }
    })
  } else {
    this.doSaveBill(newBill)
  }
}
```

```javascript
function calculateNextDue(recurring) {
  const today = new Date()
  let nextDue = new Date(today)
  
  switch (recurring.frequency) {
    case 'daily':
      nextDue.setDate(today.getDate() + 1)
      break
      
    case 'weekly':
      const currentDay = today.getDay()
      const targetDay = recurring.weeklyDay
      let daysUntil = targetDay - currentDay
      if (daysUntil <= 0) daysUntil += 7
      nextDue.setDate(today.getDate() + daysUntil)
      break
      
    case 'monthly':
      const targetMonthDay = recurring.monthlyDay
      nextDue.setDate(targetMonthDay)
      if (nextDue <= today) {
        nextDue.setMonth(nextDue.getMonth() + 1)
      }
      break
      
    case 'yearly':
      nextDue.setMonth(recurring.yearlyMonth - 1)
      nextDue.setDate(recurring.yearlyDay)
      if (nextDue <= today) {
        nextDue.setFullYear(nextDue.getFullYear() + 1)
      }
      break
  }
  
  return nextDue.toISOString().split('T')[0]
}
```

```javascript
function exportToCSV(bills, filename) {
  const headers = ['日期', '时间', '类型', '分类', '金额', '账户', '备注']
  
  const rows = bills.map(bill => [
    bill.date,
    bill.time || '',
    bill.type === 'income' ? '收入' : '支出',
    bill.category,
    bill.amount,
    bill.account,
    bill.notes || ''
  ])
  
  // 转CSV格式
  let csvContent = '\uFEFF' + headers.join(',') + '\n'  // BOM for Excel
  rows.forEach(row => {
    csvContent += row.map(cell => `"${cell}"`).join(',') + '\n'
  })
  
  // 保存文件
  const fs = wx.getFileSystemManager()
  const filePath = `${wx.env.USER_DATA_PATH}/${filename}.csv`
  
  fs.writeFile({
    filePath,
    data: csvContent,
    encoding: 'utf8',
    success: () => {
      wx.saveFileToDisk({
        filePath,
        success: () => {
          wx.showToast({ title: '导出成功', icon: 'success' })
```

```javascript
// utils/bill-helper.js
module.exports = {
  /**
   * 按时间段筛选账单
   */
  filterByPeriod(bills, period) {
    if (!Array.isArray(bills)) return []
    
    const now = new Date()
    let startDate
    
    switch (period) {
      case 'week':
        startDate = new Date(now)
        startDate.setDate(now.getDate() - 6)
        break
      case 'month':
        startDate = new Date(now.getFullYear(), now.getMonth(), 1)
        break
      case 'quarter':
        const quarter = Math.floor(now.getMonth() / 3)
        startDate = new Date(now.getFullYear(), quarter * 3, 1)
        break
      case 'year':
        startDate = new Date(now.getFullYear(), 0, 1)
        break
      default:
        return bills
    }
    
    startDate.setHours(0, 0, 0, 0)
    const todayStr = now.toISOString().split('T')[0]
    
    return bills.filter(bill => {
      if (!bill.date) return false
      return bill.date >= startDate.toISOString().split('T')[0] && bill.date <= todayStr
    })
  },

  /**
   * 按日期分组
   */
  groupByDate(bills) {
    const grouped = {}
    
    bills.forEach(bill => {
      if (!bill.date) return
      if (!grouped[bill.date]) {
        grouped[bill.date] = []
      }
      grouped[bill.date].push(bill)
    })
    
    return Object.keys(grouped)
      .sort((a, b) => b.localeCompare(a))
      .map(date => ({
        date,
        bills: grouped[date]
      }))
  },

  /**
   * 计算收支汇总
   */
  calculateSummary(bills) {
    let expense = 0
    let income = 0
    
    bills.forEach(bill => {
      const amount = parseFloat(bill.amount) || 0
      if (bill.type === 'expense') {
        expense += amount
      } else {
        income += amount
      }
    })
    
    return {
      expense: parseFloat(expense.toFixed(2)),
      income: parseFloat(income.toFixed(2)),
      balance: parseFloat((income - expense).toFixed(2))
    }
  },

  /**
   * 检测重复账单
   */
  detectDuplicate(newBill, existingBills) {
    return existingBills.find(b =>
      b.date === newBill.date &&
      b.category === newBill.category &&
      b.amount === newBill.amount &&
      b.account === newBill.account
    )
  },

  /**
   * 计算账户余额
   */
  calculateAccountBalance(bills, accountName) {
    let income = 0
    let expense = 0
    
    bills.forEach(bill => {
      if (bill.account !== accountName) return
      const amount = parseFloat(bill.amount) || 0
      if (bill.type === 'income') {
        income += amount
      } else {
        expense += amount
      }
    })
    
    return {
      balance: income - expense,
      income,
      expense
    }
  }
}
```

```javascript
// utils/formatter.js
module.exports = {
  /**
   * 格式化金额
   */
  formatMoney(amount, showSign = false) {
    const num = parseFloat(amount) || 0
    const formatted = num.toLocaleString('zh-CN', {
      minimumFractionDigits: 2,
      maximumFractionDigits: 2
    })
    return showSign ? `¥${formatted}` : formatted
  },

  /**
   * 格式化日期
   */
  formatDate(date, format = 'YYYY-MM-DD') {
    if (!date) return ''
    const d = new Date(date)
    
    const year = d.getFullYear()
    const month = String(d.getMonth() + 1).padStart(2, '0')
    const day = String(d.getDate()).padStart(2, '0')
    
    switch (format) {
      case 'MM-DD':
        return `${month}-${day}`
      case 'YYYY/MM/DD':
        return `${year}/${month}/${day}`
      case 'MM月DD日':
        return `${parseInt(month)}月${parseInt(day)}日`
      default:
        return `${year}-${month}-${day}`
    }
  },

  /**
   * 相对日期显示
   */
  formatRelativeDate(date) {
    if (!date) return ''
    
    const now = new Date()
    const d = new Date(date)
    const diff = now - d
    const days = Math.floor(diff / (1000 * 60 * 60 * 24))
    
    if (days === 0) return '今天'
    if (days === 1) return '昨天'
    if (days < 7) return `${days}天前`
    if (days < 30) return `${Math.floor(days / 7)}周前`
    
    return this.formatDate(date, 'MM-DD')
  },

  /**
   * 格式化星期
   */
  formatWeekday(date) {
    if (!date) return ''
    const weekdays = ['周日', '周一', '周二', '周三', '周四', '周五', '周六']
    const d = new Date(date)
    return weekdays[d.getDay()]
  },

  /**
   * 格式化百分比
   */
  formatPercent(value, decimals = 1) {
    const num = parseFloat(value) || 0
    return `${num.toFixed(decimals)}%`
  }
}
```
