```javascript
onQuickSave() {
  const { amount, category, account, quickType } = this.data
  if (!amount || !category) {
    wx.showToast({ title: '请输入金额和选择分类', icon: 'none' })
    return
  }
  
  // 构建账单对象

    id: Date.now(),
    type: quickType || 'expense',
    amount: parseFloat(amount),
    category,
    account: account || wx.getStorageSync('lastUsedAccount') || '现金',
    date: this.formatDate(new Date()),
    time: this.formatTime(new Date()),
    notes: this.data.quickNotes || '',
    createTime: new Date().toISOString()
  }
  
  // 保存账单
  const bills = wx.getStorageSync('bills') || []
  bills.unshift(bill)
  wx.setStorageSync('bills', bills)
  
  // 使缓存失效，确保首页刷新
  const app = getApp()
  app.invalidateCache()
  
  // 记录最后使用的账户
  wx.setStorageSync('lastUsedAccount', bill.account)
  
  wx.showToast({ title: '记账成功', icon: 'success' })
  this.resetQuickForm()
}
```

```javascript
async _processVoice(tempFilePath) {
  wx.showLoading({ title: '识别中...' })
  
  try {
    const result = await new Promise((resolve, reject) => {
      wx.translateVoice({
        localId: tempFilePath,
        isShowProgressTips: 0,
        success: resolve,
        fail: reject
      })
    })
    
    const recognizedText = result.translateResult
    console.log('语音转文字结果:', recognizedText)
    
    if (recognizedText) {
      // 解析并填充表单
      const billInfo = this._extractBillInfoFromText(recognizedText)
      this._applyVoiceResult(recognizedText, billInfo)
    } else {
      wx.showToast({ title: '未能识别语音', icon: 'none' })
    }
  } catch (err) {
    wx.showToast({ title: '识别失败', icon: 'none' })
  } finally {
    wx.hideLoading()
    this.setData({ voiceStatus: 'idle' })
  }
}
```

### 4.3 账户自动余额计算 ⭐🔥

#### 功能描述
📌 根据账单流水实时计算余额，**无需用户手动维护**，消除了数据不一致风险。

#### 用户痛点与解决方案

| 痛点 | 解决方案 | 成果 |
|------|---------|------|
| 用户需要手动维护余额 | 根据账单流水自动计算 | 📌 **余额 100% 准确** |
| 修改账单后余额不同步 | 账单变更实时触发重算 | ⭐ **实时同步** |
| 总资产需要手动汇总 | 自动汇总总资产/负债/净资产 | ⭐ **一键查看** |

#### 技术实现

```javascript
📌 // bill-helper.js - 余额计算核心逻辑
function calculateAccountBalances(bills, accounts) {
  const balanceMap = {}
  
  // 初始化所有账户
  accounts.forEach(acc => {
    balanceMap[acc.name] = {
      ...acc,
      balance: 0,
      totalIncome: 0,
      totalExpense: 0,
      transactionCount: 0
    }
  })
  
  // 遍历账单累加
  bills.forEach(bill => {
    const acc = balanceMap[bill.account]
    if (!acc) return
    
    if (bill.type === 'expense') {
      acc.balance -= bill.amount
      acc.totalExpense += bill.amount
    } else {
      acc.balance += bill.amount
      acc.totalIncome += bill.amount
    }
    acc.transactionCount++
  })
  
  // 计算汇总数据
  let totalAssets = 0
  let totalLiabilities = 0
  
  Object.values(balanceMap).forEach(acc => {
    if (acc.type === 'asset') {
      totalAssets += acc.balance
    } else {
      totalLiabilities += Math.abs(acc.balance)
    }
  })
  
  return {
    accounts: Object.values(balanceMap),
    summary: {
      totalAssets,
      totalLiabilities,
      netAssets: totalAssets - totalLiabilities
    }
  }
}
```

#### 下次到期计算

```javascript
📌 // bill-helper.js
function calculateNextDue(recurring) {
  const today = new Date()
  let nextDue = new Date(today)
  
  switch (recurring.frequency) {
    case 'daily':
      nextDue.setDate(today.getDate() + 1)
      break
    case 'weekly':
      const targetDay = recurring.weeklyDay || 0
      const currentDay = today.getDay()
      const daysUntilTarget = (targetDay - currentDay + 7) % 7 || 7
      nextDue.setDate(today.getDate() + daysUntilTarget)
      break
    case 'monthly':
      const targetDate = recurring.monthlyDay || today.getDate()
      nextDue.setMonth(today.getMonth() + 1)
      nextDue.setDate(Math.min(targetDate, daysInMonth(nextDue)))
      break
    case 'yearly':
      nextDue.setFullYear(today.getFullYear() + 1)
      break
  }
  
  return nextDue.toISOString().split('T')[0]
}
```
